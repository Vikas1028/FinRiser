--1) 
CREATE TABLE Clients (
    ClientId VARCHAR(10) PRIMARY KEY,
    Password VARCHAR(20),
    Name VARCHAR(50),
    Contact VARCHAR(10),
    EKYC TIMESTAMP,
    LastLogin TIMESTAMP,
    Suspended BOOLEAN
);


--2) 
CREATE TABLE UsersDetails(
    UserId VARCHAR(10) PRIMARY KEY,
    Adhar VARCHAR(12),
    PAN VARCHAR(10),
    FatherName VARCHAR(50),
    Address VARCHAR(200),
    AccountOpenDate TIMESTAMP
);


--3)
CREATE TABLE Dealers(
    DealerId VARCHAR(10) PRIMARY KEY,
    Password VARCHAR(20),
    Name VARCHAR(50),
    Contact VARCHAR(10),
    EKYC TIMESTAMP,
    LastLogin TIMESTAMP,
    Suspended BOOLEAN,
    Ranking INT
);


--4) 
CREATE TABLE ClientHoldings(
    ClientId VARCHAR(10) PRIMARY KEY,
    ScripId VARCHAR(50),
    Qty INT,
    BuyAvg INT 
);


--5) 
CREATE TABLE UserFund(
    UserId VARCHAR(10) PRIMARY KEY,
    AvailableCash INT,
    UtilizedCash INT
);


--6)
CREATE TABLE OrderDetails(
    OrderId VARCHAR(50) PRIMARY KEY,
    UserId VARCHAR(10),
    Type BOOLEAN,
    Qty INT,
    DateTime TIMESTAMP,
    ClientId VARCHAR(50),
    ScripId VARCHAR(50),
    Price INT
);


--7) 
CREATE TABLE Trade(
    TradeId VARCHAR(50) PRIMARY KEY,
    OrderId VARCHAR(50),
    Type BOOLEAN,
    Qty INT,
    DateTime TIMESTAMP,
    ClientId VARCHAR(50),
    ScripId VARCHAR(50),
    Price INT
);


--8) 
CREATE TABLE Position(
    UserId VARCHAR(10) PRIMARY KEY,
    Type BOOLEAN,
    Qty INT,
    DateTime TIMESTAMP,
    ClientId VARCHAR(10),
    ScripId VARCHAR(50),
    TradeId VARCHAR(50),
    OrderId VARCHAR(50) 
);
